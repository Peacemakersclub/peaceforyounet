    <script>
        let currentType = 'story';
        const urlParams = new URLSearchParams(window.location.search);
        const typeParam = urlParams.get('type');
        if (typeParam) setType(typeParam);

        function setType(type) {
            currentType = type;
            document.querySelectorAll('.type-btn').forEach(btn => {
                btn.classList.remove('bg-blue-600', 'text-white');
                btn.classList.add('bg-gray-200');
            });
            event.target.classList.remove('bg-gray-200');
            event.target.classList.add('bg-blue-600', 'text-white');
            document.getElementById('futureField').classList.toggle('hidden', type !== 'future');
        }

        function submitContent(e) {
            e.preventDefault();
            
            const content = {
                id: Date.now().toString(),
                type: currentType,
                title: document.getElementById('title').value || 'Untitled',
                content: document.getElementById('content').value,
                futureYear: document.getElementById('futureYear')?.value,
                author: {
                    name: document.getElementById('authorName').value || 'Anonymous',
                    email: document.getElementById('authorEmail').value,
                    country: document.getElementById('authorCountry').value || 'Unknown'
                },
                timestamp: new Date().toISOString(),
                reactions: { heart: 0 }
            };

            if (!content.content) {
                alert('Please write something');
                return;
            }

            const allContent = JSON.parse(localStorage.getItem('peace_content') || '[]');
            allContent.push(content);
            localStorage.setItem('peace_content', JSON.stringify(allContent));

            if (content.author.email) {
                const users = JSON.parse(localStorage.getItem('peace_users') || '{}');
                if (!users[content.author.email]) {
                    users[content.author.email] = { contributions: [], points: 0, badges: [] };
                }
                users[content.author.email].contributions.push(content.id);
                users[content.author.email].points += 10;
                if (users[content.author.email].contributions.length === 1) {
                    users[content.author.email].badges.push('?? First Voice');
                }
                localStorage.setItem('peace_users', JSON.stringify(users));
            }

            document.getElementById('thankYouModal').classList.remove('hidden');
            document.getElementById('thankYouModal').classList.add('flex');
            document.getElementById('shareForm').reset();
            return false;
        }

        function closeThankYou() {
            document.getElementById('thankYouModal').classList.add('hidden');
            document.getElementById('thankYouModal').classList.remove('flex');
        }
    </script>
